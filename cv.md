function brightest(rgb) {
let colorIndex = 0,
maxValue = 0
for (let i = 0; i < rgb.length; i++) {
let color = rgb[i],
r = parseInt(color.substring(1,3), 16),
g = parseInt(color.substring(3,5), 16),
b = parseInt(color.substring(5,7), 16),
value = Math.max(r,g,b)
if (value > maxValue) {
maxValue = value
colorIndex = i
}
}
return rgb[colorIndex]
}
